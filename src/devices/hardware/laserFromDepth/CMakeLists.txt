yarp_prepare_plugin(laserFromDepth
  CATEGORY device
  TYPE LaserFromDepth
  INCLUDE laserFromDepth.h
)

if(NOT SKIP_laserFromDepth)
  yarp_add_plugin(yarp_laserFromDepth)

  target_sources(yarp_laserFromDepth
    PRIVATE
      laserFromDepth.h
      laserFromDepth.cpp
  )

  target_link_libraries(yarp_laserFromDepth
    PRIVATE
      YARP::YARP_os
      YARP::YARP_sig
      YARP::YARP_dev
  )
  list(APPEND YARP_${YARP_PLUGIN_MASTER}_PRIVATE_DEPS
    YARP_os
    YARP_sig
    YARP_dev
  )

  yarp_install(
    TARGETS yarp_laserFromDepth
    EXPORT YARP_${YARP_PLUGIN_MASTER}
    COMPONENT ${YARP_PLUGIN_MASTER}
    LIBRARY DESTINATION ${YARP_DYNAMIC_PLUGINS_INSTALL_DIR}
    ARCHIVE DESTINATION ${YARP_STATIC_PLUGINS_INSTALL_DIR}
    YARP_INI DESTINATION ${YARP_PLUGIN_MANIFESTS_INSTALL_DIR}
  )

  set(YARP_${YARP_PLUGIN_MASTER}_PRIVATE_DEPS ${YARP_${YARP_PLUGIN_MASTER}_PRIVATE_DEPS} PARENT_SCOPE)

  set_property(TARGET yarp_laserFromDepth PROPERTY FOLDER "Plugins/Device")
endif()
